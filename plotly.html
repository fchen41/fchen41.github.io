Plotly

This page is to create a flexdashboard using plotly for that includes at least three distinct plot types. I am going to focus on the dataset from NYC Restaurant Inspections.

library(tidyverse)
library(p8105.datasets)
library(flexdashboard)
library(plotly)
data(rest_inspec)
# Clean the data. 
rest_insp_base = 
  rest_inspec %>% 
  filter(cuisine_description == "Asian", 
         !is.na(grade), 
         critical_flag == "Critical", 
         grade != "Not Yet Graded"
         ) %>% 
  select(dba, inspection_date, phone, score, street, grade, violation_code) %>%
  arrange(dba)

# Make plotly scatterplot
rest_insp_scatter =
  rest_insp_base %>%
  mutate(text_label = str_c("Cuisine: ", dba, "\nGrade: ", grade)) %>% 
  plot_ly(
    x = ~inspection_date, y = ~score, type = "scatter", mode = "markers",
    color = ~grade, text = ~text_label, alpha = 0.5)
rest_insp_scatter 
# Make a plotly boxplot
rest_insp_box = 
  rest_insp_base %>% 
  mutate(dba = fct_reorder(dba, score)) %>% 
  plot_ly(y = ~score, color = ~dba, type = "box", colors = "viridis")
rest_insp_box
# Make a plotly barchart
rest_insp_bar = 
  rest_insp_base %>% 
  group_by(grade) %>% 
  summarize(mean = mean(score)) %>% 
  plot_ly(x = ~grade, y = ~mean, color = ~mean, type = "bar", colors = "viridis")
